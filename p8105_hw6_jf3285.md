bootstrap_regression = 
weather_df %>% 
  modelr::bootstrap(n = 5000) %>% 
  mutate(
    models = map(strap, ~lm(tmax ~ tmin, data = .x) ),
    tidied = map(models, broom::tidy),
    glanced = map(models, glance)) %>% 
  select(-strap, -models) 

bootstrap_beta = 
  bootstrap_regression %>% 
  select(-glanced) %>% 
  unnest(tidied) 


bootstrap_rsq = 
  bootstrap_regression %>%
